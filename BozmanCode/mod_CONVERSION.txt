Public Function CloseForms()
Call CleanTables

Dim access_obj As AccessObject
For Each access_obj In Application.CurrentProject.AllForms
            
            If access_obj.IsLoaded = True Then
                
                If Not access_obj.name = "foreclosure database screen" And _
                Not access_obj.name = "SearchFiles" And _
                Not access_obj.name = "Main" Then
                    
                    DoCmd.Close acForm, access_obj.name
                    
                End If
            End If
        Next access_obj
    
For Each access_obj In Application.CurrentProject.AllReports
    If access_obj.IsLoaded = True Then
        DoCmd.Close acReport, access_obj.name
    End If
Next access_obj
  
End Function
