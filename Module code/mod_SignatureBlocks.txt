Function VarTrusteeSelection()


strsql = "Select * from jurisdiction where county = " & """" & Forms![foreclosure database screen]!Propertycounty & """" & " and state = " & """" & Forms![foreclosure database screen]!PropertyState & """" & ";"
Set rs = CurrentDb.OpenRecordset(strsql, dbOpenSnapshot)

If rs.EOF Then
MsgBox ("No Trustee Input for " & Forms![foreclosure database screen]!Propertycounty & " for the State: " & Forms![foreclosure database screen]!PropertyState)
Exit Function
Else

If IsNull(Forms![foreclosure database screen]![Trustee].Form![trustee2]) Then
VarTrustee = 1


Else

Dim Vt1, Vt2, Vt3 As String
Vt1 = rs![Trustee_First] & " " & rs![Trustee]
Vt2 = rs!trustee2_first & " " & rs![trustee2]
If IsNull(rs!trustee3) Then
Vt3 = ""
Else
Vt3 = rs!trustee3_first & " " & rs!trustee3
End If

If IsNull(rs!trustee3) Then
VarTrustee = InputBox("There is more than 1 Trustee for this Jurisdiction: " & vbCr & _
"Please Select " & vbCr & "1. " & Vt1 _
& " or " & vbCr & "2. " & Vt2 _
& " or " & vbCr & "3. BOTH")

Else
VarTrustee = InputBox("There is more than 1 Trustee for this Jurisdiction: " & vbCr & _
"Please Select " & vbCr & "1. " & Vt1 _
& " or " & vbCr & "2. " & Vt2 _
& " or " & vbCr & "3. " & Vt3 _
& " or " & vbCr & "4. BOTH")
End If

End If
End If

rs.Close



End Function
